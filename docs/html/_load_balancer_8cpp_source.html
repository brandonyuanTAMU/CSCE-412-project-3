100void LoadBalancer::removeServer() {
101 for (size_t i = 0; i < servers.size(); i++) {
102 if (servers[i].getAvailability()) {
103 servers.erase(servers.begin() + i);
104 serversRemoved++;
105 if ((int)servers.size() < minServers) minServers = servers.size();
106 logEvent(CLR_YELLOW, "Server removed, total servers: " + std::to_string(servers.size()));
107 return;
108 }
109 }
110 logEvent(CLR_YELLOW, "Couldn't remove a server (all busy)");
111}

136void LoadBalancer::addRequest(const Request& req) {
137 for (const std::string& blocked : blockedIPs) {
138 if (req.getIpIn().rfind(blocked, 0) == 0) { // check for prefix
139 totalRequestsBlocked++;
140 logEvent(CLR_RED, "Blocked request from IP: " + req.getIpIn());
141 return;
142 }
143 }
144 totalRequestsReceived++;
145 requests.push(req);
146 if ((int)requests.size() > maxQueueSize) maxQueueSize = requests.size();
147}

158void LoadBalancer::tick() {
159 ++time;
160 for (WebServer& server : servers) {
161 if (!server.getAvailability()) {
162 bool finished = server.tick();
163 if (finished) {
164 totalRequestsProcessed++;
165 logEvent("Request completed at time " + std::to_string(time));
166 }
167 }
168 }
169 if (time - lastAdjustedTime >= adjustCooldown) {
170 if (requests.size() > 80 * servers.size()) {
171 addServer();
172 lastAdjustedTime = time;
173 } else if (requests.size() < 50 * servers.size() && servers.size() > 1) {
174 removeServer();
175 lastAdjustedTime = time;
176 }
177 }
178 assignRequests();
179}
