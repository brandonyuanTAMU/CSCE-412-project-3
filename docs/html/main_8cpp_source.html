62void generateMasterQueue(std::vector<Request>& masterQueue, int totalServers, int duration) {
63 int initialQueueSize = totalServers * 100; // spec: "usually servers * 100"
64
65 // Phase 1: initial full queue — all arrive at t=1
66 for (int i = 0; i < initialQueueSize; i++) {
67 int processingTime = 10 + rand() % 50; // 10–60 ticks
68 char type = (rand() % 2 == 0) ? 'P' : 'S';
69 masterQueue.push_back(Request(1, processingTime, randomIP(), randomIP(), type));
70 }
71
72 // Phase 2: ongoing traffic with sine-wave modulated arrival rate
73 //
74 // Math: with N servers each taking avg 35 ticks, drain rate ≈ N/35 requests/tick.
75 // To trigger scaling, we need the queue to swing across the 50*N and 80*N thresholds.
76 // During peaks: arrival rate > drain rate → queue grows → triggers addServer
77 // During troughs: arrival rate < drain rate → queue shrinks → triggers removeServer
78 //
79 // We use a rate that swings between 0 (complete quiet) and ~N/10 (overwhelming)
80 // so that peaks strongly outpace servers and troughs let servers fully drain the queue.
81 for (int t = 2; t <= duration; t++) {
82 // Sine wave oscillates between -1 and 1; we use the RAW sine (not clamped to positive)
83 // so that ~half the cycle has zero or near-zero arrivals (quiet period)
84 double wave = sin(2.0 * M_PI * t / (duration / 4.0)) // 4 major cycles
85 + 0.5 * sin(2.0 * M_PI * t / (duration / 9.0)); // 9 smaller ripples
86
87 // Map wave to arrival rate:
88 // wave ranges from -1.5 to 1.5
89 // rate = max(0, wave) * scale → arrivals only when wave > 0 (roughly half the time)
90 double rate = std::max(0.0, wave) * (totalServers / 8.0);
91
92 // Poisson-approximate: integer part + probabilistic fractional part
93 int numRequests = (int)rate;
94 if ((rand() % 1000) / 1000.0 < (rate - numRequests)) numRequests++;
95
96 for (int i = 0; i < numRequests; i++) {
97 int processingTime = 10 + rand() % 50;
98 char type = (rand() % 2 == 0) ? 'P' : 'S';
99 masterQueue.push_back(Request(t, processingTime, randomIP(), randomIP(), type));
100 }
101 }
102}

114 srand(time(nullptr));

122 std::string logfile = "log.txt";
123 std::vector<std::string> blockedIPs;
124
125 // Parse configuration file (key=value format)
126 std::string line;
127 while (std::getline(config, line)) {
128 std::istringstream ss(line);
129 std::string key, value;
130 if (std::getline(ss, key, '=') && std::getline(ss, value)) {
131 if (key == "servers_p") serversP = std::stoi(value);
132 else if (key == "servers_s") serversS = std::stoi(value);
133 else if (key == "duration") duration = std::stoi(value);
134 else if (key == "cooldown") cooldown = std::stoi(value);
135 else if (key == "logfile") logfile = value;
136 else if (key == "blocked") {
137 std::istringstream ips(value);
138 std::string ip;
139 while (std::getline(ips, ip, ',')) {
140 blockedIPs.push_back(ip);
141 }
142 }
143 }
144 }
145
146 // Create switch with Processing and Streaming load balancers
147 Switch sw(serversP, serversS, cooldown, logfile, blockedIPs);
148
149 // Generate master queue of all requests with arrival times
150 int totalServers = serversP + serversS;
151 std::vector<Request> masterQueue;
152 generateMasterQueue(masterQueue, totalServers, duration);
153
154 // Main simulation loop: inject requests at their arrival times and tick
155 for (int t = 1; t <= duration; t++) {
156 for (const Request& req : masterQueue) {
157 if (req.getArrivalTime() == t) {
158 sw.addRequest(req);
159 }
160 }
161 sw.tick();
162 }
163 sw.end();
164 return 0;
165}
